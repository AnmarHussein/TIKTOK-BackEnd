Create Table FOLLOWER(
 ID NUMBER GENERATED BY DEFAULT AS IDENTITY CONSTRAINT PK_FOLLOWER PRIMARY Key,
 userid NUMBER CONSTRAINT FK_FOLLOWER_USERID REFERENCES USER1(ID) ON DELETE CASCADE,
 FOLLOWED NUMBER CONSTRAINT FK_FOLLOWER_FOLLOWED_USER REFERENCES USER1(ID) ON DELETE CASCADE,
 createat date,
 CONSTRAINT UN_FOLLOWER UNIQUE(userid,FOLLOWED)
);

CREATE OR REPLACE PACKAGE  FOLLOWER_PACKAGE AS

    PROCEDURE FOLLOWER_CRUD(P_ACTION in varchar DEFAULT NULL 
    ,P_ID IN NUMBER DEFAULT NULL
    ,P_userid IN NUMBER DEFAULT NULL
    ,P_FOLLOWED IN NUMBER DEFAULT NULL
    ,P_createat IN DATE DEFAULT NULL);
    
End FOLLOWER_PACKAGE;



CREATE OR REPLACE PACKAGE BODY FOLLOWER_PACKAGE AS
    
    PROCEDURE FOLLOWER_CRUD(P_ACTION in varchar DEFAULT NULL 
    ,P_ID IN NUMBER DEFAULT NULL
    ,P_userid IN NUMBER DEFAULT NULL
    ,P_FOLLOWED IN NUMBER DEFAULT NULL
    ,P_createat IN DATE DEFAULT NULL)
    IS
        follow_all SYS_REFCURSOR;
    BEGIN
        EXECUTE IMMEDIATE 'ALTER SESSION SET NLS_DATE_FORMAT=''DD-MON-YYYY HH24:MI:SS''';
        IF P_ACTION = 'INSERT' THEN
            INSERT INTO FOLLOWER VALUES(DEFAULT,P_userid,P_FOLLOWED ,To_DATE(P_createat,'DD-MM-YYYY HH24:MI:SS')); 
            COMMIT;
        ELSIF P_ACTION = 'UPDATE' THEN

            UPDATE FOLLOWER SET
                USERID = P_userid ,
                FOLLOWED = P_FOLLOWED,
                createat = To_DATE(P_createat,'DD-MM-YYYY HH24:MI:SS')
            WHERE ID = P_ID;
            COMMIT;

        ELSIF P_ACTION = 'DELETE' THEN
            DELETE FOLLOWER WHERE ID = P_ID;
            COMMIT;

        ELSIF P_ACTION = 'GETALL' THEN
            OPEN follow_all FOR SELECT * FROM FOLLOWER;
            dbms_sql.return_result(follow_all);

        ELSIF P_ACTION = 'GETBYID' THEN
            OPEN follow_all FOR SELECT * FROM FOLLOWER WHERE ID = P_ID;
            dbms_sql.return_result(follow_all); 
        END IF;
    END FOLLOWER_CRUD;

END FOLLOWER_PACKAGE;
