Create Table block1(
 ID NUMBER GENERATED BY DEFAULT AS IDENTITY CONSTRAINT PK_block1 PRIMARY Key,
 userid NUMBER CONSTRAINT FK_block1_USERID REFERENCES USER1(ID) ON DELETE CASCADE,
 block1user NUMBER CONSTRAINT FK_block1_block1_USER REFERENCES USER1(ID) ON DELETE CASCADE,
 createat date,
 CONSTRAINT UN_block1 UNIQUE(userid,block1user)
);

CREATE OR REPLACE PACKAGE  block1_PACKAGE AS

    PROCEDURE block1_CRUD(P_ACTION in varchar DEFAULT NULL 
    ,P_ID IN NUMBER DEFAULT NULL
    ,P_userid IN NUMBER DEFAULT NULL
    ,P_block1user IN NUMBER DEFAULT NULL
    ,P_createat IN DATE DEFAULT NULL);
    
End block1_PACKAGE;



CREATE OR REPLACE PACKAGE BODY block1_PACKAGE AS
    
    PROCEDURE block1_CRUD(P_ACTION in varchar DEFAULT NULL 
    ,P_ID IN NUMBER DEFAULT NULL
    ,P_userid IN NUMBER DEFAULT NULL
    ,P_block1user IN NUMBER DEFAULT NULL
    ,P_createat IN DATE DEFAULT NULL)
    IS
        block1_all SYS_REFCURSOR;
    BEGIN
        EXECUTE IMMEDIATE 'ALTER SESSION SET NLS_DATE_FORMAT=''DD-MON-YYYY HH24:MI:SS''';
        IF P_ACTION = 'INSERT' THEN
            INSERT INTO block1 VALUES(DEFAULT,P_userid,P_block1user ,To_DATE(P_createat,'DD-MM-YYYY HH24:MI:SS')); 
            COMMIT;
        ELSIF P_ACTION = 'UPDATE' THEN

            UPDATE block1 SET
                USERID = P_userid ,
                block1user = P_block1user,
                createat = To_DATE(P_createat,'DD-MM-YYYY HH24:MI:SS')
            WHERE ID = P_ID;
            COMMIT;

        ELSIF P_ACTION = 'DELETE' THEN
            DELETE block1 WHERE ID = P_ID;
            COMMIT;

        ELSIF P_ACTION = 'GETALL' THEN
            OPEN block1_all FOR SELECT * FROM block1;
            dbms_sql.return_result(block1_all);

        ELSIF P_ACTION = 'GETBYID' THEN
            OPEN block1_all FOR SELECT * FROM block1 WHERE ID = P_ID;
            dbms_sql.return_result(block1_all); 
        END IF;
    END block1_CRUD;

END block1_PACKAGE;
