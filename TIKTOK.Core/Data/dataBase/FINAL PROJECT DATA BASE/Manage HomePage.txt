
Create Table HomePage(

create table Link1(
ID NUMBER GENERATED BY DEFAULT AS IDENTITY CONSTRAINT PK_Link1 PRIMARY Key,
name Varchar2(255) not null,
urlPath Varchar2(255) not null,
isNav number(1) DEFAULT 0 not null,
homeid number Constraint fk_home_like1 REFERENCES HomePage(id) on delete Cascade
